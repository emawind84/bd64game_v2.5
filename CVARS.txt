script 851 ENTER
{
 int buttons;

 while (TRUE)
 {
	buttons = GetPlayerInput(-1, INPUT_BUTTONS);

if ((buttons & BT_SPEED) && (buttons & BT_FORWARD) && !(buttons & BT_CROUCH))
{
	GiveInventory("IsRunning", 1);
	delay(1);
}
	delay(1);
	TakeInventory("IsRunning", 1);
 }
}
